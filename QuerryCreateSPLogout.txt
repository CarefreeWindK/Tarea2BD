CREATE PROCEDURE logout
	@inNombreUsuario VARCHAR(32)
	, @inIpUsuario VARCHAR (32)
AS
BEGIN
	
	DECLARE @DATOS_JSON VARCHAR(64)='{"TipoAccion":"Logout","Description":"''"}'
	DECLARE @FechaHoraActual DATETIME
    	SET @FechaHoraActual = GETDATE() --Obtiene la fecha y hora actuales
	SET NOCOUNT ON;
	
	SELECT @IdUsuario = A.Id
	FROM dbo.Usuario A
	WHERE A.UserName = @inNombreUsuario;
	
      INSERT dbo.EventLog (
          LogDescription
          , PostIdUser
          , PostIP
          , PostTime     
      )
      VALUES (
          @DATOS_JSON
          , @IdUsuario
          , @inIpUsuario
          , @FechaHoraActual
      );
	SET NOCOUNT OFF; 
END
